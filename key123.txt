local Players = game:GetService("Players")
local CoreGui = game:GetService("CoreGui")
local StarterGui = game:GetService("StarterGui")
local HttpService = game:GetService("HttpService")

if CoreGui:FindFirstChild("SecurityGui") then
    CoreGui:FindFirstChild("SecurityGui"):Destroy()
end

local screenGui = Instance.new("ScreenGui", CoreGui)
screenGui.Name = "SecurityGui"
screenGui.ResetOnSpawn = false

local frame = Instance.new("Frame", screenGui)
frame.Size = UDim2.new(0, 350, 0, 180)
frame.Position = UDim2.new(0.5, -175, 0.4, 0)
frame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
frame.BorderSizePixel = 0
frame.BackgroundTransparency = 0.1

local label = Instance.new("TextLabel", frame)
label.Size = UDim2.new(1, -20, 0, 60)
label.Position = UDim2.new(0, 10, 0, 10)
label.Text = "Script is disabled for security reasons"
label.TextScaled = true
label.TextColor3 = Color3.fromRGB(255, 255, 255)
label.BackgroundTransparency = 1
label.Font = Enum.Font.SourceSansSemibold

local translations = {
    Spanish = "El script está desactivado por motivos de seguridad",
    English = "Script is disabled for security reasons",
    Russian = "Скрипт отключен по соображениям безопасности",
    Portuguese = "O script está desativado por motivos de segurança"
}

local idiomas = {"Spanish", "English", "Russian", "Portuguese"}

for i, idioma in ipairs(idiomas) do
    local btn = Instance.new("TextButton", frame)
    btn.Size = UDim2.new(0, 80, 0, 30)
    btn.Position = UDim2.new(0, 10 + (i - 1) * 85, 0, 80)
    btn.Text = idioma
    btn.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.Font = Enum.Font.SourceSans
    btn.TextScaled = true
    btn.MouseButton1Click:Connect(function()
        label.Text = translations[idioma]
    end)
end

local discordBtn = Instance.new("TextButton", frame)
discordBtn.Size = UDim2.new(1, -20, 0, 30)
discordBtn.Position = UDim2.new(0, 10, 1, -40)
discordBtn.Text = "Copy Discord Server Link"
discordBtn.BackgroundColor3 = Color3.fromRGB(55, 55, 255)
discordBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
discordBtn.Font = Enum.Font.SourceSansBold
discordBtn.TextScaled = true

local discordLink = "https://discord.gg/9zVNYayV"

discordBtn.MouseButton1Click:Connect(function()
    pcall(function()
        setclipboard(discordLink)
        StarterGui:SetCore("SendNotification", {
            Title = "Link copied!",
            Text = "Discord link copied to clipboard",
            Duration = 3
        })
    end)
end)
